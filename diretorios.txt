IARA/
├── .github/                  # Arquivos de comunidade para o GitHub (Issue templates, etc.)
├── .gitignore                # Ignora arquivos temporários e de compilação (essencial)
├── CONTRIBUTING.md           # Guia para novos colaboradores (como propor mudanças)
├── LICENSE                   # Licença do projeto (ex: Apache 2.0, MIT) - crucial para open source 
└── README.md                 # Cartão de visitas: instruções de instalação e visão geral

├── docs/                     # Central de toda a documentação
│   ├── academic/             # Artigos, teses e publicações científicas 
│   ├── institutional/        # Relatórios, histórico e documentos formais 
│   ├── marketing/            # Identidade visual, releases e material de divulgação
│   └── technical/            # Documentação técnica do software e hardware

├── hardware/                 # Projetos de hardware (CAD, esquemáticos) 
│   ├── electronics/          # Esquemáticos de circuitos, design de PCBs 
│   ├── mechanics/            # Arquivos CAD 3D, design estrutural, peças para impressão 
│   └── sensors/              # Datasheets e documentação específica de sensores 

├── media/                    # Ativos visuais e de mídia do projeto 
│   ├── images/               # Screenshots e materiais gráficos 
│   ├── logos/                # Logos oficiais do IARA 
│   └── videos/               # Demonstrações, apresentações em vídeo 

└── src/                      # WORKSPACE ROS 2: O coração do software do robô     │
    ├── iara_description/     # Pacote com a descrição do robô (arquivos URDF/SDF, malhas 3D)
    │
    ├── iara_simulation/      # Pacote para lançar as simulações 
    │   ├── launch/           # Arquivos de launch para Gazebo, Unreal, etc.
    │   └── worlds/           # Arquivos de mundos virtuais
    │
    ├── iara_cognition/       # Pacote para a "inteligência" e processamento cognitivo 
    │   ├── iara_llm/         # Nó para interação com Modelos de Linguagem 
    │   ├── iara_speech/      # Nós para STT (fala->texto) e TTS (texto->fala) 
    │   └── iara_emotion/     # Nó para reconhecimento e expressão de emoções 
    │
    ├── iara_perception/      # Pacote para os sistemas de percepção (visão e sensores) 
    │   ├── iara_vision/      # Nós de visão computacional: detecção de objetos, rastreamento facial 
    │   ├── iara_navigation/  # Nó para SLAM (mapeamento e localização) 
    │   └── iara_sensors/     # Nós para processar dados de sensores (táteis, proximidade, etc.)
    │
    ├── iara_control/         # Pacote para o controle motor de baixo nível 
    │   └── iara_manipulation/ # Controle de braços, mãos e garras 
    │
    └── iara_interfaces/      # Pacote para a interação com o mundo externo 
        ├── iara_ui/          # Interface de usuário, painéis de controle 
        └── iara_hri/         # Nó para a Interação Humano-Robô (HRI)
